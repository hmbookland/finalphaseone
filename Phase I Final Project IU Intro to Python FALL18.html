import pandas as pd
import numpy as np

df= pd.read_csv("C:\Users\NSTAMES\Desktop\Breast-Cancer-Wisconsin.csv", na_values= ["?"])

       'A9':'Normal Nuclei:',
       'A10':'Mitoses:',
       'CLASS':'Class:'
           }

#Use Rename( to rename the columns)
df.rename(columns=newcols, inplace=True)
    
#Rename the index## 
df.rename(index={1:'a'})

## Print Summary Statistics for All Fields##
print(df.describe(include='all'))

for x in ax:

    # Remove Spines / Despine
    x.spines['right'].set_visible(False)
    x.spines['top'].set_visible(False)
    x.spines['left'].set_visible(False)

    # Switch off ticks
    x.tick_params(axis="both", which="both", bottom="off", top="off", labelbottom="on", left="off", right="off", labelleft="on")

    vals = x.get_yticks()
    for tick in vals:
        x.axhline(y=tick, linestyle='dashed', alpha=0.4, color='#eeeeee', zorder=1)

    # Remove title
    x.set_title("")

    # Set x-axis label

    # Set y-axis label
    x.set_ylabel("Clump Thickness Instances", labelpad=20, weight='bold', size=12)

    # Format y-axis label
